390  public void Add(params object[] controllers)
391  {
392  if (controllers is null)
393  {
394  throw new ArgumentNullException(nameof(controllers));
395  }
396 
397  for (var i = 0; i < controllers.Length; i++)
398  {
399  var controller = controllers[i];
400 
401  if (controller is null)
402  {
403  throw new NullReferenceException($"Encountered a null value at index {i}.");
404  }
405 
406  var flags = BindingFlags.Public | BindingFlags.Static;
407  var type = controller as Type ?? controller.GetType();
408  if (controller is Type)
409  {
410  controller = null;
411  }
412  else
413  {
414  flags |= BindingFlags.Instance;
415  AddController(controller);
416  }
417 
418  foreach (var method in type.GetMethods(flags))
419  {
420  var route = method.GetCustomAttribute<RouteAttribute>();
421  if (route is object)
422  {
423  var parameters = method.GetParameters();
424  if (method.ReturnType == typeof(Task)
425  && parameters.Length > 0
426  && parameters.Length == route.ParameterCount
427  && parameters.Skip(1).All(p => p.ParameterType == typeof(string)))
428  {
429  var contextParamType = parameters[0].ParameterType;
430  var isHttp = contextParamType == typeof(HttpListenerContext);
431  var isWebSocket = contextParamType == typeof(ServerWebSocketConnection);
432 
433  object source = null;
434  if (!method.IsStatic)
435  {
436  source = controller;
437  }
438 
439  if (isHttp)
440  {
441  AddController(new HttpRoute(source, method, route));
442  }
443  else if (isWebSocket)
444  {
445  if (route.Authentication != AuthenticationSchemes.Anonymous)
446  {
447  throw new InvalidOperationException("WebSockets do not support authentication");
448  }
449 
450  AddController(new WebSocketRoute(source, method, route));
451  }
452  else
453  {
454  throw new InvalidOperationException($@"Method {type.Name}::{method.Name} must have a signature:
455  (System.Net.HttpListenerContext, string...) => Task
456 or
457  (Juniper.HTTP.Server.ServerWebSocketConnection, string...) => Task");
