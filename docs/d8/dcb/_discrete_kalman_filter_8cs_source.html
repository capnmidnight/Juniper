10  /// <summary>
11  /// A Kalman filter is a recursive solution to the general dynamic estimation problem for the
12  /// important special case of linear system models and Gaussian noise.
13  /// <para>The Kalman Filter uses a predictor-corrector structure, in which
14  /// if a measurement of the system is available at time <italic>t</italic>,
15  /// We first call the Predict function, to estimate the state of the system
16  /// at time <italic>t</italic>. We then call the Correct function to
17  /// correct the estimate of state, based on the noisy measurement.</para>
18  ///
19  /// <para>
20  /// The discrete Kalman filter can process linear models which have Gaussian noise.
21  /// If the model is not linear then estimate transition matrix (and other parameters if necessary) in each step and update Kalman filter.
22  /// This "dynamic" version of an Discrete Kalman filter is called Extended Kalman filter and it is used for non-linear models.
23  /// If the model is highly non-linear an Unscented Kalman filter or particle filtering is used.
24  /// See: <a href="http://en.wikipedia.org/wiki/Kalman_filter"/> for details.
25  /// </para>
26  /// </summary>
27  public class DiscreteKalmanFilter<TState, TMeasurement>

264  /// <summary>
265  /// Predicts the next state using the current state and <paramref name="controlVector"/>.
266  /// </summary>
267  /// <param name="controlVector">Set of data for external system control.</param>
268  /// <summary>
269  /// <para>Estimates the subsequent model state.</para>
270  /// <para>
271  /// x'(k) = A * x(k-1) + B * u(k).
272  /// P'(k) = A * P(k-1) * At + Q
273  /// K(k) = P'(k) * Ht * (H * P'(k) * Ht + R)^(-1)
274  /// </para>
275  /// </summary>
276  public void Predict(double[] controlVector)
277  {
278  CheckPrerequisites();
279  //x'(k) = A * x(k-1)
280  //this.state = this.state.Multiply(this.TransitionMatrix);
281  state = TransitionMatrix.Dot(state);
282 
283  //x'(k) = x'(k) + B * u(k)
284  if (controlVector is object)
285  {
286  state = state.Add(ControlMatrix.Dot(controlVector));
287  }
288 
289  //P'(k) = A * P(k-1) * At + Q
290  EstimateCovariance = TransitionMatrix.Multiply(EstimateCovariance).Multiply(TransitionMatrix.Transpose()).Add(ProcessNoise);
291 
292  /******* calculate Kalman gain **********/
293  var measurementMatrixTransponsed = MeasurementMatrix.Transpose();
294 
295  //S(k) = H * P'(k) * Ht + R
296  ResidualCovariance = MeasurementMatrix.Multiply(EstimateCovariance).Multiply(measurementMatrixTransponsed).Add(MeasurementNoise);
297  ResidualCovarianceInv = ResidualCovariance.Inverse();
298 
299  //K(k) = P'(k) * Ht * S(k)^(-1)
300  KalmanGain = EstimateCovariance.Multiply(measurementMatrixTransponsed).Multiply(ResidualCovarianceInv);
301  /******* calculate Kalman gain **********/
302  }

361  public double CalculateEntropy()
362  {
