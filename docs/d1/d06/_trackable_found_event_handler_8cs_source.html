190  /// <summary>
191  /// Check to see if the user is standing at a comfortable distance from the object.
192  /// </summary>
193  private void SafetyCheck()
194  {
195  if (Found)
196  {
197  var camT = CameraExtensions.MainCamera.transform;
198  var delta = transform.position - camT.position;
199  // we only care about the over-ground distance.
200  delta.y = 0;
201 
202  //Add 10% to the safe distance to return to safety so we don't jitter
203  // over the edge.
204  Safe = Safe && delta.magnitude >= safeDistance
205  || !Safe && delta.magnitude >= safeDistance * 1.1f;
206 
207  OnSafeOrProximity();
208  }
209  }
