471  private static IEnumerator LoadScenePathCoroutine(string scenePath, IProgress prog)
472  {
473  var sceneName = GetSceneNameFromPath(scenePath);
474  var sceneLoadProg = prog.Subdivide(0, 0.25f, sceneName + ": loading...");
475  var subSceneLoadProg = prog.Subdivide(0.25f, 0.75f, sceneName + ": loading components...");
476 
477 
478  sceneLoadProg.Report(0);
479  if (!IsSceneLoaded(scenePath))
480  {
481  if (Application.isPlaying)
482  {
483  var curPriority = Application.backgroundLoadingPriority;
484  Application.backgroundLoadingPriority = ThreadPriority.Low;
485 
486  var op = SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
487  while (!op.isDone)
488  {
489  sceneLoadProg.Report(op.progress);
490  yield return null;
491  }
492 
493  Application.backgroundLoadingPriority = curPriority;
494  }
495 #if UNITY_EDITOR
496  else
497  {
498  EditorSceneManager.OpenScene(scenePath, OpenSceneMode.Additive);
499  }
500 #endif
501  while (!IsSceneLoaded(scenePath))
502  {
503  yield return null;
504  }
505  }
506  sceneLoadProg.Report(1);
507 
508  var scene = SceneManager.GetSceneByPath(scenePath);
509 
510  if (Application.isPlaying)
511  {
512  var toLoad = scene.FindAll<SubSceneController>().ToArray();
513  for (var i = 0; i < toLoad.Length; ++i)
514  {
515  var ss = toLoad[i];
516  if (!ss.CanEnter && ss.CanExit)
517  {
518  yield return ss.ExitAsync().AsCoroutine();
519  }
520 
521  if (ss.CanEnter)
522  {
523  yield return ss.EnterAsync(subSceneLoadProg.Subdivide(i, toLoad.Length, ss.name)).AsCoroutine();
524  }
525  }
526  }
